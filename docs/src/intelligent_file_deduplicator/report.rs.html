1use std::{
2    fs::File,
3    io::Write,
4    path::Path,
5    collections::HashMap,
6};
7use serde::Serialize;
8
9#[derive(Serialize)]
10struct DuplicateGroup {
11    hash: String,
12    files: Vec<String>,
13}
14
15#[derive(Serialize)]
16struct Report {
17    duplicates: Vec<DuplicateGroup>,
18}
19
20pub fn write_json_report<P: AsRef<Path>>(
21    hash_map: &HashMap<String, Vec<String>>,
22    output_path: P,
23) -> std::io::Result<()> {
24    // A vector of `DuplicateGroup` structs, where each group represents a set of files
25    // that share the same hash (i.e., are duplicates).
26    //
27    // This vector is constructed by filtering the entries of a hash map to include only
28    // those where multiple files share the same hash, and then mapping each such entry
29    // to a `DuplicateGroup` containing the hash and the list of duplicate files.
30    //
31    // Example:
32    // // Given a hash_map of file hashes to file paths:
33    // let duplicates: Vec<DuplicateGroup> = hash_map
34    //     .iter()
35    //     .filter(|(_, files)| files.len() > 1)
36    //     .map(|(hash, files)| DuplicateGroup {
37    //         hash: hash.clone(),
38    //         files: files.clone(),
39    //     })
40    //     .collect();
41    // // `duplicates` now contains groups of files with identical content.
42    let duplicates: Vec<DuplicateGroup> = hash_map
43        .iter()
44        .filter(|(_, files)| files.len() > 1)
45        .map(|(hash, files)| DuplicateGroup {
46            hash: hash.clone(),
47            files: files.clone(),
48        })
49        .collect();
50
51    let report = Report { duplicates };
52    let json = serde_json::to_string_pretty(&report).expect("Serialization failed");
53
54    let mut file = File::create(output_path)?;
55    file.write_all(json.as_bytes())?;
56
57    Ok(())
58}
